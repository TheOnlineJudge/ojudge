MACRO (FILE_TO_STRING infile outfile var)
ADD_CUSTOM_COMMAND(OUTPUT ${outfile}
                   COMMAND ${CMAKE_COMMAND}
                   ARGS -Dinfile:FILEPATH=xml/${infile}
                        -Doutfile:FILEPATH=${outfile}
                        -Dvar:STRING=${var}
                        -P filetostring.cmake
                   MAIN_DEPENDENCY
                        xml/${infile}
)
ENDMACRO (FILE_TO_STRING)
